## File 1: .github/workflows/whatsapp_bot.yml

```yaml
name: WhatsApp Islamic Post Sender

on:
  schedule:
    - cron: '0 14 * * *'
  workflow_dispatch:

jobs:
  send-posts:
    runs-on: ubuntu-latest
    
    steps:
    - name: Checkout code
      uses: actions/checkout@v4
      
    - name: Set up Python
      uses: actions/setup-python@v5
      with:
        python-version: '3.11'
        
    - name: Install dependencies
      run: |
        python -m pip install --upgrade pip
        pip install requests schedule
        
    - name: Run WhatsApp Bot
      env:
        ULTRAMSG_TOKEN: ${{ secrets.ULTRAMSG_TOKEN }}
        ULTRAMSG_INSTANCE: ${{ secrets.ULTRAMSG_INSTANCE }}
      run: |
        python whatsapp_bot_cloud.py --now
        
    - name: Commit folder renames
      continue-on-error: true
      run: |
        git config --global user.name "WhatsApp Bot"
        git config --global user.email "bot@users.noreply.github.com"
        git add -A
        git diff --staged --quiet || git commit -m "Auto: Mark posts as sent [skip ci]"
        git push || echo "No changes to commit"
      env:
        GITHUB_TOKEN: ${{ secrets.GITHUB_TOKEN }}
```
